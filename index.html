} // ----------------------------------------------------------------- // A. 根據分數區間改變文本顏色和內容 (畫面反映一) // ----------------------------------------------------------------- // 確保文本清晰：在文字周圍繪製一個半透明的白色矩形 fill(255, 200); } } // ================================================================= // 步驟三：定義煙火和粒子類別 (用於特效) // ----------------------------------------------------------------- class Particle { constructor(x, y, hu, firework) { this.pos = createVector(x, y); // 隨機方向 this.vel.mult(random(2, 10)); // 隨機速度 } this.acc = createVector(0, 0); } applyForce(force) { // 施加的力與粒子的質量無關，因此直接相加 this.acc.add(force); } update() { if (!this.firework) { this.vel.mult(0.9); // 隨機顏色 this.firework = new Particle(x, y, this.hu, true); // 火箭粒子 this.exploded = false; this.particles = []; } update() { if (!this.exploded) { this.firework.applyForce(gravity); this.firework.update(); // 爆炸條件：當火箭開始下墜時 (垂直速度由負轉正) if (this.firework.vel.y >= 0) { this.exploded = true; this.explode(); } } for (let i = this.particles.length - 1; i >= 0; i--) { this.particles[i].applyForce(gravity); this.particles[i].update(); if (this.particles[i].done()) { this.particles.splice(i, 1); } } } explode() { for (let i = 0; i < 100; i++) { // 生成 100 個粒子 const p = new Particle(this.firework.pos.x, this.firework.pos.y, this.hu, false); this.particles.push(p); } } show() { if (!this.exploded) { this.firework.show(); } for (const p of this.particles) { p.show(); } } done() { // 如果已經爆炸且所有粒子都消失了，則煙火結束 return this.exploded && this.particles.length === 0; } }
